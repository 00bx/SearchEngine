ناوی بابەت :
چۆن لووپ لەگەڵ داتایێکی زۆر بەکاردێنیت؟



مەبەستی ئەم پرسیارە ئەوەیە کە ئەگەر بێت و کۆمەڵێک داتا و ئایدی زۆرت هەبێت مەسەلەن دەتەوێ ئەڤرێجی دەرەجەی ١٠٠ قوتابی لە مەکتەبێکدا بخەیتە فیلدێکی نوێ بەناوی ئەڤرێجەوە لە تەیبلێک بەناوی ستودێن_مارک ، کە تەیبلەکە بۆ هەر قوتابیێک دەرەجەی هەر دەرسێکی تێدا دانراوە، ڕەنگە بتوانیت ڤاریەبلێک دروست بکەیت و بە وایڵ لووپەکە دروست بکەین وە ڤاڕیەبلەکە ١ تێدا داخل بکەین و بچووکتری بکەین لە ١٠٠ بەم شێوەیە

declare i default 1

while i <= 100 do

update from student_marks

set average_mark = (select avg(mark) from student_mark group by student_id )

where student_id = i

set i = i+1

end while


ئەمە وەڵامێکی تەواوە بەڵام ئەگەری دروست بوونی کێشەی زۆرە مەسەلەن قوتابیێک ئایدیەکەی ٢٣ ، پێش ەو قوتابیە ئایدی ٢٠ هەبووە کەواتە فیلدی ٢١ و ٢٢ بەتاڵن ، ئەمەش ئیررۆر دروستەکات یاخود ئەگەر هاتوو لە سیستەمەک لەشوێنی ئایدی کۆدی نیشتمانی نەتەوەیەک پێوست بێت لە سیستەمەکە ، ئێ ئەم ئایدیییە لە ١ەوە دەستپێناکات و کۆتایشی دیار نییە پێوستمان بە ڕێگەیەکی دیەکەیە بۆ چارەسەر

: وەڵامی ڕاست


SET @counter = 1;

SET @num_students = (select max(student_id));

لێرە لە شوێنی ئەوەی خۆمان کۆتا ئایدی بنووسین ، ئەگریگەیت فەنکشنی ماکس بەکاردێنین


WHILE @counter <= @num_students DO

update from student_marks

set average_mark = (select avg(marks) )

بۆ دۆزینەوەی ئەڤرێجەکە و خستنە ناوەوەی بۆ فیلدی ئەڤرێج فیلدەکە یەکسان دەکەی بەو کلاوسەی کە ئەڤرێج مارکمان بۆ دەدۆزێتەوە بۆ هەر قوتابییەک کە بۆ نموونە لەو تەیبلە دەرەجەی ٣ دەرسی تێدایە هەرسێ دەرسەکە دەرەجەکانیان کۆ دەکرێتەوە دابەش ٣ دەکرێت بۆ هەر قوتابیەک


where student_id = @counter;

لێرەش ئایدیەکە یەکسان دەکەین بە ڤاریەبلی کاونتەر کە لە ١ەوە دەستپێدەکات تا دەگاتە کۆتا قوتابی


SET @counter = @counter + 1;

END WHILE;

: ئەنجام


لەبەرەوەی دەرەجەی یەک دەرس ئەڤرێجی دەردێت ،دابەشی ١ی دەکات ئەگریگەیت فەنکشنەکە بۆیە ئەنجامی ئەڤرێجەکە وەک دەرەجەکە دەبێت ، بەڵام ئەگەر دوو دەرسبێت کۆی هەردوو دەرسەکە دەکات و دابەش دوویان دەکات ئەگریگەیت فەنکشنەکە پاشان ئەنجامەکە دەخاتە ناو فیلدەکەوە ، دەتوانی بۆ هەر ئایدیێک دەرسێکی تر زیاد بکەیت هەر ئەم کویریەش بەکاربێنەوە بۆ دڵنیابوون لە وەڵامەکە
